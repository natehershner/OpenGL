#version 330 core

layout (location=0) in vec3 vertexPos;

out vec2 fragmentTexCoord;

uniform mat4 model;

const vec3 colors[4] = vec3[](
	vec3(1.0, 0.0, 0.0),
	vec3(0.0, 1.0, 0.0),
	vec3(0.0, 0.0, 1.0),
	vec3(0.5, 0.5, 0.0)
);

void main() {
	gl_Position = model * vec4(vertexPos, 1.0);
	fragmentTexCoord = vec2(0.5, -0.5) * (vertexPos.xy + vec2(1.0));
}
